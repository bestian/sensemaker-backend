[
  {
    "id": "comment-1",
    "text": "這個產品真的很棒，使用起來非常方便！",
    "votes": {
      "agrees": 15,
      "disagrees": 2,
      "passes": 1
    }
  },
  {
    "id": "comment-2", 
    "text": "界面設計很美觀，功能也很實用。",
    "votes": {
      "agrees": 12,
      "disagrees": 1,
      "passes": 0
    }
  }
]

comment-id,comment_text,agrees,disagrees,passes
